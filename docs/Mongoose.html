Mongoose

1. 安裝Mongoose​

  • 安裝Mongoose:
npm install --save mongoose

2. 連線至資料庫​

  • 與MongoDB建立連線:
// ./app.js
const mongoose = require('mongoose');

mongoose.connect('mongodb://username:userpassword@host:port/database', {
  useNewUrlParser: true
})  // 連線至資料庫

const db = mongoose.connection;  // 取得連線狀態
db.on('error', (err) => console.error('connection error', err));  // 連線異常
db.once('open', (db) => console.log('Connected to MongoDB'));     // 連線成功

3. 建立Schema​

  • 建立Schema:
// 以practice.js為例
// ./modules/practice.js
const mongoose = require('mongoose');

let PracticeSchema = new mongoose.Schema(
    {
        code: {
            type: String,   // 資料型態
            required: true, // required表示必須輸入，未輸入會回報錯誤
            enum: ["LEARNER", "WRITER", "EDITOR", "ADMIN"]  // enum表示限制輸入內容，輸入其他東西會回報錯誤
        },
        status: {
            type: Boolean,
            default: true   // default表示預設值，未賦值時會輸入該值
        }
    },
    {
        timestamps: true,   // timestamps為true時會自動增加createdAt及updatedAt欄位
        versionKey: false   // versionKey為false時不會顯示version欄位
    },
);

PracticeSchema.set('collection', 'practice_for_James');         // 將Schema連至資料庫
module.exports = mongoose.model('practice', PracticeSchema);    // 將Schema編譯成Model

4. CRUD​

  • 先引入Model:
const practice = require('./modules/practice');
  • Create:
const doc1 = new practice(  // 建立doc1
  {
    code: "LEARNER",
    status: true
  }
);
const doc2 = new practice( // 建立doc2
  {
    code: "ADMIN",
    status: false
  }
);
const docs = [doc1, doc2]; // 彙集成docs

practice.insertMany(docs)  // 新增至資料庫
  .then(function () {
    console.log("Created!");
  })
  .catch(function (err) {
    console.log(err);
  });
  • Read:
async function Find(){
  const A = await practice.find({status: true}); // 括號內可設定尋找條件
  console.log(A);
}

Find();
  • Update:
practice.updateOne({code:"ADMIN"}, {$set: {status: false}}) // updateOne({欲更改項目}, {$set: {欲更改數值}})
  .then(function () {
    console.log("Updated!");
  })
  .catch(function (err) {
    console.log(err);
  });
  • Delete:
practice.deleteMany({code: "EDITOR"})  // deleteMany({欲刪除項目})
  .then(function () {
    console.log("Deleted!");
  })
  .catch(function (err) {
    console.log(err);
  });
