設定Schema​

// Model.js
const mongoose = require('mongoose');

let PracticeSchema = new mongoose.Schema(
    {
        code: {
            type: String,
            required: true,
            enum: ["LEARNER", "WRITER", "EDITOR", "ADMIN"]
        },
        status: {
            type: Boolean,
            default: true
        }
    },
    {
        timestamps: true,
        versionKey: false
    },
);
PracticeSchema.set('collection', 'practice_for_James');
module.exports = mongoose.model('practice', PracticeSchema);

設定Routes​

// Routes.js
module.exports = function(app) {
    var todoList = require('../controllers/Controller');

    // Routes
    app.route('/tasks') // 處理GET/POST
        .get(todoList.list_all_tasks)
        .post(todoList.create_a_task);

    app.route('/tasks/:taskId') // 處理 GET/PUT/DELETE
        .get(todoList.read_a_task)
        .put(todoList.update_a_task)
        .delete(todoList.delete_a_task);
};

設定Controller​

// Controller.js
'use strict';

var mongoose = require('mongoose'),
    Task = mongoose.model('practice');  //一開始建立的Tasks schema

//針對不同request的回應(json格式)

exports.list_all_tasks = function(req, res) {
    Task.find({}, function(err, task) {
        if (err)
            res.send(err);
        res.json(task);
    });
};

//CRUD

exports.create_a_task = function(req, res) {
    var new_task = new Task(req.body);
    new_task.save(function(err, task) {
        if (err)
            res.send(err);
        res.json(task);
    });
};

exports.read_a_task = function(req, res) {
    Task.findById(req.params.taskId, function(err, task) {
        if (err)
            res.send(err);
        res.json(task);
    });
};

exports.update_a_task = function(req, res) {
    Task.findOneAndUpdate(req.params.taskId, req.body, { new: true }, function(err, task) {
        if (err)
            res.send(err);
        res.json(task);
    });
};

exports.delete_a_task = function(req, res) {
    Task.remove({
        _id: req.params.taskId
    }, function(err, task) {
        if (err)
            res.send(err);
        res.json({ message: 'Task successfully deleted' });
    });
};

3. 操作Postman​

  1. 選擇GET並輸入http://localhost:3000/tasks即可查看所有資料
  2. 動作改為POST並在Body輸入對應的Key跟Value即可新增資料
  3. 動作改為DELETE並講網址改為http://localhost:3000/tasks/taskid即可將資料刪除
  4. 此時動作改為GET將得到單筆資料
